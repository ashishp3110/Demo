SELECT ALL SALARY.EMPID, SALARY.EMPNAME, SALARY.DESIGID, SALARY.DESIGNAME, 
SALARY.SECTID, SALARY.SECTNAME, SALARY.DEPTID, SALARY.DEPTNAME, SALARY.SECTIONID, 
SALARY.SECTIONNAME, TO_CHAR(TO_DATE(SALARY.PERIOD, 'YYYYMM'), 'MONTH,YYYY') PER,
((NVL(SALARY.MDAYS,0))-(NVL(SALARY.ADAYS,0))) PAYDAY, 
SALARY.ABASIC, SALARY.AHRENT, SALARY.AMEDI, SALARY.AATTN, 
SALARY.OVRHOURS, SALARY.OVRRATE, (NVL(SALARY.OVRHOURS, 0)*NVL(SALARY.OVRRATE, 0)) TOTAT, 
SALARY.ABSDEDUC, SALARY.ABONUS, (((NVL(SALARY.ABASIC, 0))+(NVL(SALARY.AHRENT, 0))+(NVL(SALARY.AMEDI, 0))+(NVL(SALARY.AATTN, 0)))+((NVL(SALARY.OVRHOURS, 0))*(NVL(SALARY.OVRRATE, 0)))) GROSS, 
((((NVL(SALARY.ABASIC, 0))+(NVL(SALARY.AHRENT, 0))+(NVL(SALARY.AMEDI, 0))+(NVL(SALARY.AATTN, 0)))+((NVL(SALARY.OVRHOURS, 0))*(NVL(SALARY.OVRRATE, 0))))-(NVL(SALARY.ABSDEDUC, 0))) NPAY
FROM SALARY
WHERE (SALARY.PERIOD ='200304') &PMWHERE
